# 構成イメージ（Pythonスクリプト）

import feedparser
import anthropic

# 1. RSSから記事を取得
feeds = [
    "https://hnrss.org/frontpage",
    "https://dev.to/feed",
]

articles = []
for feed_url in feeds:
    feed = feedparser.parse(feed_url)
    for entry in feed.entries[:5]:  # 各フィードから5件
        articles.append({
            "title": entry.title,
            "url": entry.link,
            "summary": entry.get("summary", "")
        })

# 2. Claude APIで要約・フィルタリング
client = anthropic.Anthropic()

for article in articles:
    message = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1000,
        messages=[{
            "role": "user",
            "content": f"""
以下の記事をITエンジニア向けに評価してください。

タイトル：{article['title']}
内容：{article['summary']}

以下の形式で回答してください：
- ブログ記事化の価値：高/中/低
- 日本語での要約（100字以内）
- アフィリエイトと紐づけられる案件があれば提案
            """
        }]
    )
    article["evaluation"] = message.content[0].text

# 3. 結果をNotionまたはSlackに送信

Claude Codeとは何か
Claude Codeはターミナルから直接AIにコーディングを依頼できるツールで、コード補完・開発支援が主な用途です。

Claude Code  → エンジニアがコードを書くための道具

Claude APIが必要であり、Claude Codeは関係しない
